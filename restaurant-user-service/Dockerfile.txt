# Dockerfile in restaurant-user-service

# Use official Node.js image
FROM node:18

# Set working directory
WORKDIR /app

# Copy package files and install dependencies
COPY package*.json ./
RUN npm install

# Copy the rest of the code
COPY . .

# Build TypeScript (if you're using TS)
RUN npm run build

# Expose service port
EXPOSE 3001

# Start the service
CMD ["npm", "run", "start"]
